void insert_bottom(int x) {
if (top == -1) {
push(x);
return;
}
int temp = pop();
insert_bottom(x);
push(temp);
}
void reverse() {
if (top == -1)
return;
int temp = pop();
reverse();
insert_bottom(temp);
}
